📋 Cómo configurar Google Sheets (5 pasos)
  1. Abra Google Sheets y cree una hoja nueva
  2. Vaya a Extensiones → Apps Script
  3. Borre todo el contenido y pegue el código de abajo
  4. Haga clic en Implementar → Nueva implementación → Aplicación web
      → En "Quién tiene acceso" elija "Cualquier persona"
  5. Copie la URL generada y péguela en "Configurar URL" de arriba
Código para pegar en Apps Script:
// ============================================================
// APPS SCRIPT — Poder Judicial del Perú
// Pegar en: Google Sheets → Extensiones → Apps Script
// ============================================================

var HOJA_DATOS    = 'PJ_Datos';
var HOJA_LOG      = 'PJ_Log';

function doGet(e) {
  try {
    var ss    = SpreadsheetApp.getActiveSpreadsheet();
    var hoja  = ss.getSheetByName(HOJA_DATOS) || ss.insertSheet(HOJA_DATOS);
    var raw   = hoja.getRange('A1').getValue();
    var datos = raw ? JSON.parse(raw) : { expedientes: [], audiencias: [] };
    return respuesta({ ok: true,
      expedientes: datos.expedientes || [],
      audiencias:  datos.audiencias  || [],
      total:       (datos.expedientes || []).length,
      actualizado: datos.actualizado || '' });
  } catch(e) {
    return respuesta({ ok: false, error: e.message });
  }
}

function doPost(e) {
  try {
    var ss    = SpreadsheetApp.getActiveSpreadsheet();
    var hoja  = ss.getSheetByName(HOJA_DATOS) || ss.insertSheet(HOJA_DATOS);
    var body  = JSON.parse(e.postData.contents);
    var accion = body.accion || 'escribir';

    if (accion === 'escribir') {
      var payload = {
        expedientes: body.expedientes || [],
        audiencias:  body.audiencias  || [],
        actualizado: new Date().toLocaleString('es-PE'),
        usuario:     body.usuario || 'Anónimo'
      };
      hoja.getRange('A1').setValue(JSON.stringify(payload));
      registrarLog(ss, body.usuario, 'ESCRITURA', (body.expedientes||[]).length + ' expedientes');
      return respuesta({ ok: true, actualizado: payload.actualizado, total: payload.expedientes.length });
    }

    if (accion === 'guardar_uno') {
      var raw  = hoja.getRange('A1').getValue();
      var datos = raw ? JSON.parse(raw) : { expedientes: [], audiencias: [] };
      var exp  = body.expediente;
      var idx  = datos.expedientes.findIndex(function(e){ return e.num === exp.num; });
      if (idx >= 0) datos.expedientes[idx] = exp;
      else datos.expedientes.push(exp);
      datos.actualizado = new Date().toLocaleString('es-PE');
      datos.usuario     = body.usuario || 'Anónimo';
      hoja.getRange('A1').setValue(JSON.stringify(datos));
      registrarLog(ss, body.usuario, 'MODIFICACIÓN', exp.num);
      return respuesta({ ok: true, actualizado: datos.actualizado });
    }

    if (accion === 'eliminar_uno') {
      var raw2 = hoja.getRange('A1').getValue();
      var datos2 = raw2 ? JSON.parse(raw2) : { expedientes: [], audiencias: [] };
      var antes = datos2.expedientes.length;
      datos2.expedientes = datos2.expedientes.filter(function(e){ return e.num !== body.num; });
      datos2.actualizado = new Date().toLocaleString('es-PE');
      hoja.getRange('A1').setValue(JSON.stringify(datos2));
      registrarLog(ss, body.usuario, 'ELIMINACIÓN', body.num);
      return respuesta({ ok: true, eliminados: antes - datos2.expedientes.length });
    }

    return respuesta({ ok: false, error: 'Acción no reconocida: ' + accion });
  } catch(e) {
    return respuesta({ ok: false, error: e.message });
  }
}

function registrarLog(ss, usuario, accion, detalle) {
  try {
    var log = ss.getSheetByName(HOJA_LOG);
    if (!log) { log = ss.insertSheet(HOJA_LOG); log.appendRow(['Fecha','Usuario','Acción','Detalle']); }
    log.appendRow([new Date().toLocaleString('es-PE'), usuario||'Anónimo', accion, detalle]);
  } catch(e) {}
}

function respuesta(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
Compartir con otros usuarios: Todos deben pegar la misma URL en "Configurar URL". Los datos quedan guardados en su Google Sheets y cualquier usuario puede cargarlos o modificarlos.
